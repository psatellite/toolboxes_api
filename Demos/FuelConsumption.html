% Define altitude and Mach number
%--------------------------------
h = 25400;
m = 5;

% Compute pressure and velocity
%------------------------------
p = StdAtm(h);
v = m*p.speedOfSound;

% Scimitar thrust and specific fuel consumption
%----------------------------------------------
[t, sFC] = Scimitar( v, h );

fprintf(1,'\nSFC %e (kg/N-s)\n',sFC)

% Computes L/D ratio from Mach number
%------------------------------------
lOverD = LiftToDragFromMach( m );

% Define vector for range and dry mass
%-------------------------------------
r      = linspace(12051,15345,33);
mD     = 100000;

% Compute and display daily fuel use
%-----------------------------------
mF     = FuelFromRange( mD, lOverD, r, v, sFC );
fprintf(1,'\nDaily use of H2 %12.4f MT\n',sum(mF)/1000);
