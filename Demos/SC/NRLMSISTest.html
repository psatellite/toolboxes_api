flags.switches(10)=1; % Does not use daily ap

alt = linspace(0,2000);
rho = zeros(1,length(alt));
input(1).ap_a = aph;
for k = 1:length(alt)
  input(1).alt = alt(k);
  flags.switches(1) = 0;
	output = AtmNRLMSISE( 'gtd7', [], input(1), flags );
  rho(k) = output.d(6);
end

Plot2D(alt,rho,'Altitude (km)','\rho (gm/cm^3)','Atmosphere','ylog')
