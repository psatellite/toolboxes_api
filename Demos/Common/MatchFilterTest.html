m = zeros(1,100);
m(76:91) = 1;
k = 1:length(m);


n  = rand(1,length(m));

mN = m + n;


fM  = fft(m);
fMN = fft(mN);
fN  = fft(n);


q = ifft(fM.*fMN);
y={'Signal', 'With Noise', 'Match Filter Signal'}
Plot2D(k,[m;mN;q],'Sample',y,'Time Histories')



mT = zeros(1,100);
mT(33:40) = 1;

fMT  = fft(mT);

q = ifft(fMT.*fMN);

Plot2D(k,[m;mN;q],'Sample',y,'Time Histories Different Pulse')
