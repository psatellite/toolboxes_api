A scan of continuous burn optimizations

Assumes a constant, low acceleration. Performs many optimizations with different tradeoffs of delta-V and mission time. This demo takes multiple hours to run. It is 'off' by default for purposes of buildng the toolbox; to run it, you must first change the flag at the top.

See also: IntegrateBurnPolicyContinuous, AccelFromPolicyContinuous, PlotTrajectoryContinuous, PlotDiagnosticContinuous, OptimizePolicyContinuous

if 1
  warning('This is a very long demo; change the flag in this demo to run it.')
  return;
end

fprintf('Alert! This demo can take multiple hours to run. \n');

initialGuess = [pi pi;-0.4*pi 0.4*pi];

% Set up scan variable
costFactorMin = 10^(-3);
costFactorMax = 10^(-0);
costFactorsPerDecade = 3;
costFactors = 10.^((log(costFactorMin)/log(10)):(1/costFactorsPerDecade):(log(costFactorMax)/log(10)));
costFactors = costFactors(end:-1:1); % Reverse order

% Set up output library
guesses = zeros([size(initialGuess) numel(costFactors)]);
dVs = zeros(size(costFactors));
tDurs = zeros(size(costFactors));
boolAgrees = zeros(size(costFactors));
% Make planar after initialization
guesses2 = zeros([size(initialGuess)+[2 0] numel(costFactors)]);
dVs2 = zeros(size(costFactors));
tDurs2 = zeros(size(costFactors));
boolAgrees2 = zeros(size(costFactors));


% Set up mission parameters
dSim = IntegrateBurnPolicyContinuous; % Simulation data structure
dSim.el0 = [7000 0 0 0 0.01 0]; % Initial orbit, Planar at first, then non
dSim.el1 = [42164 0 0 0 0 0]; % Higher, circular is better
dSim.mu = 3.98600436e5;
dSim.tMax = 8e6; % Seconds, maximum mission time
dSim2 = dSim;
dSim2.el0(2) = 28*pi/180;
dAccel = AccelFromPolicyContinuous; % Acceleration data structure
dAccel.mu = dSim.mu;
dAccel.el1 = dSim.el1;
dAccel.aMag = 1e-5; % Accel. km/s/s
dAccel.Isp = []; % Isp, seconds
dAccel.tMax = dSim.tMax;
odeOpts = odeset( 'AbsTol', 1e-7, 'RelTol', 1e-5 ); % Trade simulation speed for accuracy

for i=1:numel(costFactors)
  fac=costFactors(i);
  optOpts	= optimset(...
    'maxiter',1200,...
    'MaxFunEvals',12000); % Optimization options

  [outGuess,dV,tDur,boolAgree,exitFlag,optOut] = OptimizePolicyContinuous(initialGuess,false,fac,dSim,dAccel,odeOpts,optOpts);

  % Report
  guesses(:,:,i) = outGuess;
  dVs(i) = dV;
  tDurs(i) = tDur;
  boolAgrees(i) = boolAgree;


  newGuess = [outGuess;-0.4*pi 0.4*pi;0 0]; % Initialize nonplanar guess from planar solution

  [outGuess2,dV2,tDur2,boolAgree2,exitFlag,optOut] = OptimizePolicyContinuous(newGuess,false,fac,dSim2,dAccel,odeOpts,optOpts);

  % Report
  guesses2(:,:,i) = outGuess2;
  dVs2(i) = dV2;
  tDurs2(i) = tDur2;
  boolAgrees2(i) = boolAgree2;

end

NewFig('Continous Fuel Time Opt')
plot(tDurs(dVs>0),dVs(dVs>0),'--*')
xlabel('Mission duration (s)');ylabel('Delta-V (km/s)');
title('Fuel-time optimal points for low thrust LEO to GEO at a=1e-2 m/s^2')
hold on;
[dVLTS, ~, tLTS] = LTSpiral( 7000, 42164, 1e-5);
plot(tLTS*24*60*60,dVLTS,'g*');
text(tLTS*24*60*60,dVLTS,'\leftarrow LTSpiral');
dVHoh = DVHoh(dSim.el0(1),dSim.el1(1));
dVHohI = OptDIH( 0, dSim2.el0(1), 0, dSim2.el1(1), dSim2.el0(2));
plot(tDurs(dVs>0),tDurs(dVs>0)*0+dVHohI);
text(max(tDurs(dVs>0)),dVHohI,'\leftarrow H+I');
plot(tDurs(dVs>0),tDurs(dVs>0)*0+dVHoh);
text(max(tDurs(dVs>0)),dVHoh,'\leftarrow Hohmann');


plot(tDurs2(boolAgrees2==1),dVs2(boolAgrees2==1),'--*')
