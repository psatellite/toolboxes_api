Loops

k: number of pairs of burns j: number of laps the longest of these is about 135 seconds. Mass always goes down with more burns. Note: duration goes UP from single lap to 1 pair of burns, then down as more pairs of burns are added.

for k = 1:length(nBurnPairs)
  for j = 1:length(nLaps)
    data.nBurnPairs = nBurnPairs(k);
    data.laps = nLaps(j);
    if data.nBurnPairs == 0
      % relax tolerances to keep it from running forever
      data.ConTol = 1e-3;  % 1e-6
      data.stepTol = 1e-6; % 1e-10
    end
    if nLaps(j)*nBurnPairs(k)>40
      % skip
      break;
    end
    disp('Optimizing...')
    tic
    [burnsOpt,elOutput,out] = OptimizeMassImpulsive(elInitial,elFinal,burns0,data);
    et(j,k) = toc;
    fprintf('...finished in %g sec\n', et(j,k))
    OptimizerReport( out )
    massV(j,k) = out.mS;
    daysV(j,k) = out.absTimes(end)/86400;
    dVs(j,k) = out.dVtotal;
    burns{j,k} = out.burns;
    outs{j,k} = out;
    laps(j,k) = nLaps(j);
    pairs(j,k) = nBurnPairs(k);
  end
end
