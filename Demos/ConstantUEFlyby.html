payload             = 300;  % kg
nAU                 = 600;  % number of AU
nYear               = 20;
fTank               = 0.03; % tank fraction

aU                  = Constant('au');
year                = 365*86400;

d                   = Straight2DStructure;
d.f                 = fTank;
d.mP                = payload;
d.dF                = nAU*aU;
d.eta               = 0.4;     % power to thrust efficiency
d.sigma             = 1*1000;  % W/kg
d.tF                = nYear*year;

uEOpt = FlybyOptimalImpulse( d );
d.uE  = uEOpt;
[s,data] = FlybyReport(payload,d);
DisplayLatexTable( s )

SimulateFlyby( data )
