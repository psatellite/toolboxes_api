dT = 10;
rPM = 6000;
e   = 0;
a   = 1e-9;
c   = 7.3e10;
n   = 22000000;
f   = zeros(2,n);

for k = 1:n
  e     = e + dT*rPM;
  sigma = a*exp(e/c);
  f(:,k)= [sigma;sigma*RandChiSquared(1)];
end

j = find(f>1);
fprintf('Number of failures %d\n',length(j));

t = (0:n-1)*dT;
TimeHistory(t,f,{'\sigma' 'Failure' },'Bearing Failure')
x = get(gca,'xlim');
line(x,[1 1],'color',[1 0 0]);
