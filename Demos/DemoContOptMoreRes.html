A non-impulsive optimization demo with a resolution increase step

An optimization using OptimizePolicyContinuous. Uses one optimization as the initial state of a new optimization with a higher-resolution burn policy.

See also: IntegrateBurnPolicyContinuous, AccelFromPolicyContinuous, PlotTrajectoryContinuous, PlotDiagnosticContinuous

disp('Skip by default as this demo takes many hours.')
return;

dSim = IntegrateBurnPolicyContinuous; % Simulation data struct
dAccel = AccelFromPolicyContinuous; % Acceleration data struct

% Initial and final orbits
dSim.el0 = [7000 0 0 0 0.01 0];
dSim.el1 = [42164 0 0 0 0 0];
dAccel.el1 = dSim.el1;

% Simulation time
dSim.tMax = 1e8; % Seconds
dAccel.tMax = dSim.tMax;

% Spacecraft parameters
dAccel.aMag = 1e-6; % Accel. km/s/s
dAccel.Isp = []; % Isp, seconds

% Discretized burn power angles, initial guess
initialGuess = [pi pi;-0.4*pi 0.4*pi];

% Set up optimization balance
fac = 0.005; % Takes 11 hours. 3.835 km/s, 3.16e7 seconds.

odeOpts = odeset( 'AbsTol', 1e-8, 'RelTol', 1e-6 ); % Trade simulation speed for accuracy

optOpts	= optimset(...
  'maxiter',1200,...
  'MaxFunEvals',12000); % Optimization options

% Plot initial guess
[~,dV,tDur,boolAgree] = OptimizePolicyContinuous(initialGuess,2,fac,dSim,dAccel,odeOpts,optOpts);
if ~boolAgree
  fprintf('Alert! You''ve chosen an initial condition which does not reach final state. \n');
  beep
end

% Perform the first optimization
[outGuess,dV,tDur,boolAgree,exitFlag,optOut] = OptimizePolicyContinuous(initialGuess,1,fac,dSim,dAccel,odeOpts,optOpts);

dSim2 = dSim;
dAccel2 = dAccel;
% First: recalibrate tMax so that it's closer to plausible
dSim2.tMax = min(tDur*1.1,dSim.tMax);
dAccel2.tMax = dSim2.tMax;
initialGuess2 = outGuess;
initialGuess2(:,2) = initialGuess2(:,1) + (initialGuess2(:,2) - initialGuess2(:,1))*dAccel2.tMax/dAccel.tMax;
% Next: Add an intermediate point to initialGuess2
initialGuess2 = [initialGuess2(:,1) (initialGuess2(:,1)+initialGuess2(:,2))/2 initialGuess2(:,2)];

[~,dV2,tDur2,boolAgree2] = OptimizePolicyContinuous(initialGuess2,2,fac,dSim2,dAccel2,odeOpts,optOpts);
if ~boolAgree2
  fprintf('Alert! You''ve chosen an initial condition which does not reach final state. \n');
  beep
end

% Perform the next optimization
[outGuess2,dV2,tDur2,boolAgree2,exitFlag2,optOut2] = OptimizePolicyContinuous(initialGuess2,1,fac,dSim2,dAccel2,odeOpts,optOpts);
