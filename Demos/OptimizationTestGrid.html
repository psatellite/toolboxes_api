  dT(1,k) = toc;

Gather data from different algorithms into single variables

%------------------------------------------------------------
xT = {};
xO = {};
xF = {};
for k = 1:length(name)
  xT{k,1} = name{k};
  xO{k,1} = name{k};
  xF{k,1} = name{k};
  xF{k,2} = fMin(k);
  for j = 1:size(dT,1)
    xT{k,j+1} = dT(j,k);
    xO{k,j+1} = fP(j,k);
    xF{k,j+2} = fF(j,k);
  end
end
