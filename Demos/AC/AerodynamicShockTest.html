function [nErrors, fTested] = AerodynamicShockTest( ~ )

%--------------------------------------------------------------------------
%   Test AerodynamicShock.
%--------------------------------------------------------------------------
%   Form:
%   [nErrors, fTested] = AerodynamicShockTest( ~ )
%--------------------------------------------------------------------------
%
%   ------
%   Inputs
%   ------
%   verbose 	           Print if == 1 (default 0)
%
%   -------
%   Outputs
%   -------
%   nErrors               Number of errors
%   fTested               Files tested
%
%--------------------------------------------------------------------------

%--------------------------------------------------------------------------
%   Copyright (c) 2019 Princeton Satellite Systems, Inc.
%   All rights reserved.
%--------------------------------------------------------------------------

if( nargin < 1 )
   verbose = 0;
end

AerodynamicShock;
err(1) = 0;

% Output
k       = find( err > 100*eps);
nErrors = length(k);


% List all files tested here
fTested = {'AerodynamicShock'};

% If verbose is selected print out useful results to the command window
if( nErrors > 0 && verbose )
	fprintf('\nErrors detected in tests %s of:\n',mat2str(k));
	disp(fTested);
end

%--------------------------------------

ans =

     0
