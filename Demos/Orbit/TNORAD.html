rV = NORAD( 0, 1, 5, 'sgp', [], 'sgp.txt' );
t  = [0 360 720 1080 1440];  % minutes

fprintf('\nSGP-File Input\n\n')
disp('Position')
for k = 1:5
  fprintf('%11.2f %11.2f %11.2f %11.2f\n',t(k), rV.r(:,k));
end
fprintf('\nVelocity\n')
for k = 1:5
  fprintf('%11.2f %11.2f %11.2f %11.2f\n',t(k), rV.v(:,k));
end

s = [sprintf('SGPTest \n'),...
     sprintf('1 88888U          80275.98708465  .00073094  13844-3  66816-4 0     8\n'),...
		 sprintf('2 88888  72.8435 115.9689 0086731  52.6988  110.5714 16.05824518  105')];
rV = NORAD( 0, 1, 5, 'sgp', [], s );

fprintf('\nSGP-String input\n\n')
disp('Position')
for k = 1:5
  fprintf('%11.2f %11.2f %11.2f %11.2f\n',t(k), rV.r(:,k));
end
fprintf('\nVelocity\n')
for k = 1:5
  fprintf('%11.2f %11.2f %11.2f %11.2f\n',t(k), rV.v(:,k));
end
