Demonstrate a fuel budget for the ComStar satellite.

The spacecraft has three fuel tanks. Stationkeeping events are scheduled for the lifetime of the spacecraft. There are two fuel tanks for hydrazine and one for the main engine. The output will be in the file "ArianeFBudget.txt" in OrbitData.

Since version 2.

Mission stationkeeping maneuvers

%---------------------------------
[dVEW, acc, dTEW] = DVGeoTrx( 107.6, 0.1 );
dVEW              = dVEW/dTEW;
nEW               = life/dTEW;

Schedule stationkeeping events

%-------------------------------
dEW         = 0;
dNS         = 0;
dInYr       = 365.25-31-19;
[dVNS,dTNS] = DVIDrift(box,year);

for i = 0:life
  dInYr = dInYr + 1;
  if dInYr > 365.25
    year       = year + 1;
	  dInYr      = 0;
    [dVNS,dTNS] = DVIDrift(box,year);
  end
  if  i > (dEW + dTEW)
    dEW    = i;
	  k      = k + 1;
    data(k,1:length(x)) = x;
    event(k).type          = 'delta-v';
    event(k).description   = ['East/West ', num2str(year)];
    event(k).aCSThruster   = 1:12;
    event(k).dVThruster    = [1 2 3 4];
    event(k).dVTotal       = dVEW;
    event(k).dHTotal       = [];
    event(k).dTControl     = 1;
    event(k).nPulses       = [];
    event(k).duration      = [];
    event(k).regulated     =  0;
    event(k).cM            = cM;
  end
  if  i > (dNS + dTNS)
    dNS    = i;
    k      = k + 1;
    event(k).type          = 'delta-v';
    event(k).description   = ['North/South ', num2str(year)];
    event(k).aCSThruster   = 1:12;
    event(k).dVThruster    = [13 14];
    event(k).dVTotal       = dVNS;
    event(k).dHTotal       = [];
    event(k).dTControl     = 1;
    event(k).disturbance   = [];
    event(k).nPulses       = [];
    event(k).duration      = [];
    event(k).regulated     =  0;
    event(k).cM            = cM;
  end

end
