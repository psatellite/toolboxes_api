Design a two stage to orbit launch vehicle

The first stage is called the Ferry and the second the Orbiter.

See also: WingedLaunchVehicleDesign, DrawVertices

dO = WingedLaunchVehicleDesign(	'name',               'Orbiter',...
                                'wing position',      'bottom',...
                                'hydrogen',           1.5732e+03,...
                                'oxygen',             9.2502e+03,...
                                'height',             2.2,...
                                'width',              2.2,...
                              	'payload length',     0.8,...
                               	'engine length',      2.22,...
                              	'engine diameter',    1.44,...
                              	'fuselage roof height',	0.5,...
                               	'tail height',    2);
dO.v(:,3) = dO.v(:,3) - 2.1;

dF = WingedLaunchVehicleDesign(	'name',           'Ferry',...
                               	'wing position',	'top',...
                               	'hydrogen',       1.5732e+03,...
                                'height',         1.5,...
                                'width',          2.2,...
                                'payload length', 6,...
                               	'inlet height',   0.5,...
                              	'inlet angle',    60,...
                                'tail separation', 3,...
                              	'tail angle',      20,...
                                'tail height',    2);


f   = dF.f;
fO  = dO.f + size(dF.v,1);
v   = [dF.v;dO.v];
f   = [f;fO];

v = dF.v;
f = dF.f;

DrawVertices( v, f, 'SRT', true );
