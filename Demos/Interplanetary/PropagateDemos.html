PropagateCRTBPBackwards

s       = load('f16pRef.mat');
y       = s.d.initialState';
t0      = 0;
tF      = s.d.period;
[t,y]   = PropagateCRTBPBackwards(y,t0,tF);
[t,tL]  = TimeLabl(t);
yL      = {'x' 'y' 'z' 'v_x' 'v_y' 'v_z'};
Plot2D(t,y,tL,yL,'PropagateCRTBP');
